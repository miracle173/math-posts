def lex_max_matrix(A):
    best = None

    # alle Permutationen der Zeilen (0,1,2)
    for rp in permutations(range(3)):
        # Zeilen permutiert
        R = [A[i] for i in rp]

        # alle Permutationen der Spalten

            M = [[row[j] for j in cp] for row in R]
